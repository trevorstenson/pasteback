Rust

pub fn normalize(input: &str) -> String {
    input
        .lines()
        .map(str::trim_end)
        .filter(|line| !line.is_empty())
        .collect::<Vec<_>>()
        .join("\n")
}

SQL

WITH recent_captures AS (
  SELECT id, source_app, created_at
  FROM captures
  WHERE created_at > now() - interval '7 days'
)
SELECT source_app, count(*) AS total
FROM recent_captures
GROUP BY source_app
ORDER BY total DESC;
